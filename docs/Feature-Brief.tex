\section{Core Commerce Features}

\subsection{Ordering Flow (Pieces or Liters)}
\begin{featurebox}{Feature Overview}
\textbf{What:} Order by pieces (boxes) or liters with real-time totals, delivery fee threshold, validation, and a structured delivery form.

\vspace{0.3cm}
\textbf{Why:} Supports retail carton flows and bulk ordering.

\vspace{0.3cm}
\textbf{User Value:} Clear pricing/minimums; fast, trustworthy checkout.

\vspace{0.3cm}
\textbf{Company Value:} Fewer manual corrections; higher conversion; controllable unit economics.
\end{featurebox}

\begin{kpibox}
\textbf{KPIs:} Checkout conversion, Average Order Value (AOV), repeat order rate, validation drop-offs.
\end{kpibox}

\subsection{Delivery Tracking}
\begin{featurebox}{Feature Overview}
\textbf{What:} Real-time order tracking with live status updates (Order Confirmed, Preparing, Out for Delivery, Delivered), delivery personnel details, estimated arrival time, and live map showing delivery location. Push notifications for each status change.

\vspace{0.3cm}
\textbf{Why:} Reduces delivery anxiety and support inquiries by providing transparency throughout the fulfillment process.

\vspace{0.3cm}
\textbf{User Value:} Peace of mind knowing exactly when their order will arrive; ability to plan accordingly.

\vspace{0.3cm}
\textbf{Company Value:} Reduced "Where is my order?" support tickets; improved delivery success rates; higher customer satisfaction.
\end{featurebox}

\begin{kpibox}
\textbf{KPIs:} Tracking page views per order, delivery success rate, support ticket reduction, customer satisfaction score (CSAT) for delivery, on-time delivery rate.
\end{kpibox}

\subsection{Payment Methods \& Preferences}
\begin{featurebox}{Feature Overview}
\textbf{What:} Settings area for payment methods and user preferences .

\vspace{0.3cm}
\textbf{Why:} Faster repeat checkout; fewer payment failures.

\vspace{0.3cm}
\textbf{User Value:} Saved methods and consistent defaults.

\vspace{0.3cm}
\textbf{Company Value:} Higher repeat rate; lower friction and support load.
\end{featurebox}

\begin{kpibox}
\textbf{KPIs:} Saved-method penetration, repeat checkout time, payment failure rate.
\end{kpibox}

\subsection{Customer Care, Help Center, About}
\begin{featurebox}{Feature Overview}
\textbf{What:} Customer Care modal, Help Center entry, and About modal within Settings.

\vspace{0.3cm}
\textbf{Why:} Industrial buyers expect clear support and self-service.

\vspace{0.3cm}
\textbf{User Value:} Faster answers and resolution.

\vspace{0.3cm}
\textbf{Company Value:} Lower churn; deflection of simple tickets.
\end{featurebox}

\begin{kpibox}
\textbf{KPIs:} Customer Satisfaction Score (CSAT), First Response Time, Time to Resolution, deflection rate.
\end{kpibox}

\subsection{Flavor Ratings \& Reviews}

\vspace{0.3cm}

\subsection{Store Reviews (Location-level)}
\begin{featurebox}{Feature Overview}
\textbf{What:} Reviews for store/outlet locations in addition to products.

\vspace{0.3cm}
\textbf{Why:} Surfaces service quality and operational issues by location.

\vspace{0.3cm}

\subsection{Community Feed}
\begin{featurebox}{Feature Overview}
\textbf{What:} A brand-safe social feed with a composer, rich post cards, and interactions. Post types include Flavor Reviews, Challenge Completed badges, photo posts, and stories with hashtags. Users can like, comment, and share.

\vspace{0.3cm}
\textbf{Why:} Turns authentic user moments into ongoing social proof and distribution, amplifying core funnels.

\vspace{0.3cm}
\textbf{User Value:} Easy way to share experiences, celebrate challenge completions, discover trending flavors, and interact with the community.

\vspace{0.3cm}
\textbf{Company Value:} Continuous UGC pipeline, increased reach via shares, and behavioral signals to inform merchandising and campaigns.
\end{featurebox}

\begin{kpibox}
\textbf{KPIs:} DAU/WAU on feed, posts/day, engagement rate (likes+comments+shares), share-out rate, CTR from feed to flavor/review/challenge, moderation pass rate.
\end{kpibox}

\textbf{\color{subsectionTeal}Notes:} Brand-safe moderation and reporting/flagging, spam/abuse controls, optional tag filters, and privacy controls supported.

\section{Loyalty \& Rewards}

\subsection{Wallet \& Loyalty (IceCoins UI)}
\begin{featurebox}{Feature Overview}
\textbf{What:} A complete loyalty hub with:
\begin{itemize}[leftmargin=*]
    \item Wallet home card showing balance with "Earn More" and "Redeem" CTAs
    \item Recent Activity ledger with positive/negative entries
    \item Available Rewards gallery and Redeem Rewards list with tile cards
    \item Prototype accrual/burn from purchases, challenges, promos; redemption confirmation via toast
\end{itemize}

\vspace{0.3cm}
\textbf{Why:} Makes loyalty tangible by showing value, history, and clear ways to redeem—driving retention and frequency.

\vspace{0.3cm}
\textbf{User Value:} Know exactly how many coins they have, where they came from, and what they can get—redeem in a tap.

\vspace{0.3cm}
\textbf{Company Value:} Increases repeat rate and ARPU; creates room for targeted promos and cross-feature ties.
\end{featurebox}

\begin{kpibox}
\textbf{KPIs:} Wallet adoption, redemption rate, time-to-first redemption, earn:burn ratio, outstanding liability, promo uplift, share of accrual from challenges.
\end{kpibox}

\textbf{\color{subsectionTeal}Notes:} In production, keep a server-side ledger, anti-abuse checks, expiration policy, and dispute workflow. Integrates with Promotions \& Ads Center for campaigns.

\subsection{Challenges \& Missions (IceCoins)}
\begin{featurebox}{Feature Overview}
\textbf{What:} A goal-based challenge system with progress tracking and IceCoins rewards. Examples:
\begin{itemize}[leftmargin=*]
    \item "Flavor Explorer — Try 5 new flavors this month"
    \item "Social Star — Share 3 ice cream photos"
    \item "Eco Warrior — Use your own cup 5 times"
\end{itemize}
Supports time-bounded windows (weekly/monthly), per-challenge progress bars, and instant reward issuance upon completion.

\vspace{0.3cm}
\textbf{Why:} Nudge high-value behaviors across trial, content creation, and sustainability.

\vspace{0.3cm}
\textbf{User Value:} Clear goals, visible progress, and meaningful rewards.

\vspace{0.3cm}
\textbf{Company Value:} Increased SKU trial, steady UGC supply, and adoption of eco behaviors.
\end{featurebox}

\begin{kpibox}
\textbf{KPIs:} Challenge opt-in rate, completion rate, average IceCoins issued/user, lift to conversion post-completion, UGC submissions approved, eco action completions, abuse detection rate.
\end{kpibox}

\subsection{Referral / Pyramid Rewards}
\begin{featurebox}{Feature Overview}
\textbf{What:} Invite flow rewarding both referrer and referee; optional tiered bonuses.

\vspace{0.3cm}
\textbf{Why:} Harnesses word-of-mouth at scale.

\vspace{0.3cm}
\textbf{User Value:} Earn rewards by inviting friends.

\vspace{0.3cm}
\textbf{Company Value:} Low-CAC acquisition channel.
\end{featurebox}

\begin{kpibox}
\textbf{KPIs:} Invites sent, acceptance rate, k-factor, cost per acquired user.
\end{kpibox}

\subsection{Promotions \& Ads Center}
\begin{featurebox}{Feature Overview}
\textbf{What:} Centralized, brand-safe promotional surfaces for campaigns such as "Double IceCoins Week." Supports hero banners, feed cards, and section headers with gradients/imagery. Campaigns can deep-link to wallet, offers, products, or challenges.

\vspace{0.3cm}
\textbf{Why:} A scalable way to broadcast offers and seasonal initiatives without app updates.

\vspace{0.3cm}
\textbf{User Value:} Clear visibility of active promotions and easy one-tap access.

\vspace{0.3cm}
\textbf{Company Value:} Higher reach and measurable uplift for campaigns; flexible targeting and timing.
\end{featurebox}

\begin{kpibox}
\textbf{KPIs:} Banner viewable impressions, CTR, assist to conversion, incremental revenue vs. baseline, frequency cap hit rate, creative A/B test winner rate.
\end{kpibox}

\section{Health \& Sustainability}

\subsection{Health \& Nutrition}
\begin{featurebox}{Feature Overview}
\textbf{What:} A dedicated hub for health-conscious shopping including:
\begin{itemize}[leftmargin=*]
    \item Health Profile (weight, height, auto-calculated BMI)
    \item Dietary preferences and allergies
    \item Daily calorie goal with progress tracking
    \item BMI Calculator
    \item Personalized Recommendations with "Perfect Match" badge
    \item Side-by-side Nutrition Compare for two flavors
    \item Clear Allergen Information badges per flavor
\end{itemize}

\vspace{0.3cm}
\textbf{Why:} Helps users quickly identify products that fit their goals and restrictions while building trust.

\vspace{0.3cm}
\textbf{User Value:} Safer choices for allergens/diets, awareness of calories and BMI, confidence via "Perfect Match."

\vspace{0.3cm}
\textbf{Company Value:} Expands addressable audience, improves recommendation relevance, reduces post-purchase dissatisfaction.
\end{featurebox}

\begin{kpibox}
\textbf{KPIs:} Health profile completion rate, BMI calculator usage, users with a set calorie goal, CTR on "Recommended for you," conversion lift for "Perfect Match" items, allergen filter usage, nutrition-compare usage.
\end{kpibox}

\textbf{\color{subsectionTeal}Notes:} Not medical advice; provide plain-language disclaimer. Respect privacy by storing only what's necessary. Integrates with Recommendations, Search/Filters, and product detail pages.

\textbf{Why:} Aligns with values-driven purchasing and sustains engagement over time.

\vspace{0.3cm}
\textbf{User Value:} Recognition for positive impact, motivation through progress and levels.

\vspace{0.3cm}
\textbf{Company Value:} Differentiation and CSR storytelling; ongoing re-engagement and brand affinity.
\end{featurebox}

\begin{kpibox}
\textbf{KPIs:} Badge unlock rate, eco views/session, return visits to tracker, level progression, correlation to retention, cohort retention.
\end{kpibox}

\textbf{User Value:} Secure access, faster repeat actions with saved profile info, and control over their data.

\vspace{0.3cm}

\begin{kpibox}

\end{kpibox}

\textbf{\color{subsectionTeal}Notes:} Production hardening includes secure password hashing, rate limiting, bot protection, email/OTP verification, and social login options. Regional compliance for consent and data retention required.
