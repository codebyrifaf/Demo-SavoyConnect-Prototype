\section{PHASE 0: Core MVP}
\begin{phasebox}{App Launch Essentials}
These features are essential for user trust, discovery, and first purchase. The app must launch with all of these.
\end{phasebox}

\subsection{Onboarding \& Sign-Up}
\begin{featurebox}{Feature Overview}
\textbf{Goal:} Convert every app installer into a registered user with zero friction.

\vspace{0.3cm}
\textbf{Core Features:}
\begin{itemize}[leftmargin=*]
    \item Phone number login with OTP verification
    \item Optional Google / Apple social login
    \item One-page intro (3 slides max): ``Explore flavors,'' ``Find nearest Savoy,'' ``Earn rewards''
\end{itemize}
\end{featurebox}

\begin{examplebox}
\textbf{User Flow Example:}\\
When first opened, user sees: ``Welcome to Savoy Ice Cream — Let's make your first order sweeter!''\\
Then a quick 3-step setup $\rightarrow$ OTP verification $\rightarrow$ Home screen with ``Welcome Offer: 10\% off first scoop.''
\end{examplebox}

\textbf{\color{subsectionTeal}Implementation Tips:} Use Firebase Auth for OTP, show animated loader while verifying. Keep UI minimal (Poppins or Nunito fonts).

\subsection{Store Locator}
\begin{featurebox}{Feature Overview}
\textbf{Goal:} Help users find the nearest Gallery quickly.

\vspace{0.3cm}
\textbf{Core Features:}
\begin{itemize}[leftmargin=*]
    \item Interactive map with Savoy location pins
    \item Each store card shows: name, address, hours, contact, directions
    \item Distance from current location
\end{itemize}
\end{featurebox}

\begin{examplebox}
\textbf{User Flow Example:}\\
User opens ``Find a Gallery'' tab $\rightarrow$ sees ``Dhanmondi Gallery (1.2 km away)'' $\rightarrow$ taps $\rightarrow$ opens Google Maps navigation.
\end{examplebox}

\textbf{\color{subsectionTeal}Implementation Tips:} Use Google Maps SDK. Store data from backend: \texttt{branch\_id}, \texttt{lat/lon}, \texttt{image}, \texttt{open\_hours}.

\subsection{Menu \& Flavours}
\begin{featurebox}{Feature Overview}
\textbf{Goal:} Let users browse visual menu to build appetite.

\vspace{0.3cm}
\textbf{Core Features:}
\begin{itemize}[leftmargin=*]
    \item Categorized menu (e.g., ``Classic Scoops,'' ``Ice Cream Cakes,'' ``Discone Specials'')
    \item Each item includes: image, price, description, allergens, availability
    \item Filters: Available today, Best sellers, Limited edition
\end{itemize}
\end{featurebox}

\begin{examplebox}
\textbf{User Flow Example:}\\
User opens ``Flavours'' tab $\rightarrow$ sees:
\begin{itemize}[leftmargin=*]
    \item Belgian Chocolate -- BDT 90/scoop
    \item Mango Mania (Seasonal) -- ``Available till December''
\end{itemize}
\end{examplebox}

\textbf{\color{subsectionTeal}Implementation Tips:} Store product catalog in backend. Use content tags for ``New,'' ``Popular,'' etc.

\subsection{Order for Pickup}
\begin{featurebox}{Feature Overview}
\textbf{Goal:} Enable customers to place pickup orders from nearest branch.

\vspace{0.3cm}
\textbf{Core Features:}
\begin{itemize}[leftmargin=*]
    \item Select branch $\rightarrow$ choose items $\rightarrow$ add to cart $\rightarrow$ choose pickup time
    \item Payment options: bKash, Nagad, Credit/Debit card
    \item Order confirmation with push notification
\end{itemize}
\end{featurebox}

\begin{examplebox}
\textbf{User Flow Example:}\\
Rafi orders ``2 scoops of Shahi Kheer + Discone.'' Pays via bKash $\rightarrow$ receives push notification: ``Ready for pickup in 15 minutes.''
\end{examplebox}

\textbf{\color{subsectionTeal}Implementation Tips:} Backend order status flow: Pending $\rightarrow$ Accepted $\rightarrow$ Ready $\rightarrow$ Completed. Integration: SSLCommerz + Firebase Cloud Messaging.

\subsection{Promotions \& Coupon Engine}
\begin{featurebox}{Feature Overview}
\textbf{Goal:} Encourage new users to buy immediately.

\vspace{0.3cm}
\textbf{Core Features:}
\begin{itemize}[leftmargin=*]
    \item Auto-apply welcome offer (10\% off first order)
    \item Promo codes like SAVOY10 or MANGOLOVE
    \item QR-based redemption for in-store offers
\end{itemize}
\end{featurebox}

\begin{examplebox}
\textbf{User Flow Example:}\\
User receives push notification: ``Flash Offer! 20\% off on Mango Mania — today only!'' Taps notification $\rightarrow$ coupon auto-applies at checkout.
\end{examplebox}

\textbf{\color{subsectionTeal}Implementation Tips:} Simple JSON-based rule engine: \{\texttt{code}, \texttt{discount\_type}, \texttt{value}, \texttt{expiry}, \texttt{conditions}\}.

\subsection{Loyalty \& Rewards (Points System)}
\begin{featurebox}{Feature Overview}
\textbf{Goal:} Build repeat purchase habit through gamified rewards.

\vspace{0.3cm}
\textbf{Core Features:}
\begin{itemize}[leftmargin=*]
    \item Earn points per purchase (e.g., 10 points for every BDT 100 spent)
    \item Visual progress bar displayed on home screen
    \item Redeem points for free scoop, toppings, or discounts
\end{itemize}
\end{featurebox}

\begin{examplebox}
\textbf{User Flow Example:}\\
User sees banner: ``You're 20 points away from a free scoop!'' This motivates them to order again to reach the milestone.
\end{examplebox}

\textbf{\color{subsectionTeal}Implementation Tips:} Store points in user profile table. Trigger reward milestone push notification after redemption.

\subsection{Push Notifications \& Messaging}
\begin{featurebox}{Feature Overview}
\textbf{Goal:} Keep users engaged and informed throughout their journey.

\vspace{0.3cm}
\textbf{Core Features:}
\begin{itemize}[leftmargin=*]
    \item \textbf{Transactional:} ``Your order is ready for pickup.''
    \item \textbf{Marketing:} ``New flavour alert — Blueberry Bliss now available!''
    \item \textbf{Personalized:} ``Your favorite Mango Mania is back!''
\end{itemize}
\end{featurebox}

\begin{examplebox}
\textbf{User Flow Example:}\\
After 5 idle days, user receives: ``Hey Rafi, your free scoop is waiting. Come claim it at Dhanmondi Gallery!''
\end{examplebox}

\textbf{\color{subsectionTeal}Implementation Tips:} Segment users via Firebase topics: \texttt{nearby\_branch}, \texttt{favourite\_flavour}, \texttt{loyal\_users}.

\subsection{Admin Dashboard (for Gallery Team)}
\begin{featurebox}{Feature Overview}
\textbf{Goal:} Let store managers control branch info, menu, and orders efficiently.

\vspace{0.3cm}
\textbf{Core Features:}
\begin{itemize}[leftmargin=*]
    \item Manage inventory and availability in real-time
    \item View live orders and update order status
    \item Push new promo notifications to customers
\end{itemize}
\end{featurebox}

\begin{examplebox}
\textbf{User Flow Example:}\\
Manager at Dhanmondi marks ``Kulfi Delight – Out of stock.'' Item instantly disappears from customer-facing menu.
\end{examplebox}

\textbf{\color{subsectionTeal}Implementation Tips:} Web admin panel built with React + Express. Implement role-based access control (manager/admin).

\section{PHASE 1: Engagement \& Personalization}
\begin{phasebox}{Growth \& Retention Features}
These features focus on user engagement, personalization, and organic growth through referrals and recommendations.
\end{phasebox}

\subsection{Referral Program}
\begin{featurebox}{Feature Overview}
\textbf{Goal:} Acquire new users through word-of-mouth marketing.

\vspace{0.3cm}
\textbf{Core Features:}
\begin{itemize}[leftmargin=*]
    \item Invite friend and both receive BDT 50 off
    \item Unique referral codes per user
    \item Track referral links and conversions
\end{itemize}
\end{featurebox}

\begin{examplebox}
\textbf{User Flow Example:}\\
Rafi shares code \texttt{RAFI50} $\rightarrow$ friend installs app $\rightarrow$ friend places order $\rightarrow$ both receive reward instantly.
\end{examplebox}

\textbf{\color{subsectionTeal}Implementation Tips:} Generate unique referral codes per user; store referral events in analytics for tracking viral coefficient.

\subsection{Personalized Recommendations}
\begin{featurebox}{Feature Overview}
\textbf{Goal:} Boost repeat orders through intelligent product suggestions.

\vspace{0.3cm}
\textbf{Core Features:}
\begin{itemize}[leftmargin=*]
    \item Suggest ``Your usual order'' based on history
    \item Recommend similar items based on preferences
    \item Display personalized homepage content
\end{itemize}
\end{featurebox}

\begin{examplebox}
\textbf{User Flow Example:}\\
User previously bought Belgian Chocolate, sees: ``You may also like: Mocha Fusion.''
\end{examplebox}

\textbf{\color{subsectionTeal}Implementation Tips:} Start with rule-based recommendations (collaborative filtering); later upgrade to ML models for better accuracy.

\subsection{User Reviews \& Ratings}
\begin{featurebox}{Feature Overview}
\textbf{Goal:} Build trust and create feedback loop for product improvement.

\vspace{0.3cm}
\textbf{Core Features:}
\begin{itemize}[leftmargin=*]
    \item Rate flavours 1–5 stars with optional text/photo
    \item Highlight top-rated items
    \item Moderate reviews for quality control
\end{itemize}
\end{featurebox}

\begin{examplebox}
\textbf{User Flow Example:}\\
Product page shows: ``Mango Mania – 4.8★ (Based on 320 reviews)''
\end{examplebox}

\textbf{\color{subsectionTeal}Implementation Tips:} Allow review submission only after verified purchase to prevent spam.

\subsection{Order History \& Reorder}
\begin{featurebox}{Feature Overview}
\textbf{Goal:} Simplify frequent purchases with one-tap reordering.

\vspace{0.3cm}
\textbf{Core Features:}
\begin{itemize}[leftmargin=*]
    \item View complete order history with details
    \item One-tap reorder button for past orders
    \item Track order status and receipts
\end{itemize}
\end{featurebox}

\begin{examplebox}
\textbf{User Flow Example:}\\
User opens ``My Orders'' tab $\rightarrow$ sees previous order $\rightarrow$ taps ``Reorder same combo'' $\rightarrow$ instantly added to cart.
\end{examplebox}

\textbf{\color{subsectionTeal}Implementation Tips:} Cache last order as JSON snapshot for quick retrieval and reordering.

\subsection{Geo-Based Push Offers}
\begin{featurebox}{Feature Overview}
\textbf{Goal:} Drive foot traffic through contextual, location-based engagement.

\vspace{0.3cm}
\textbf{Core Features:}
\begin{itemize}[leftmargin=*]
    \item Trigger notifications when user is near a branch
    \item Location-specific offers and promotions
    \item Respect user privacy with opt-in permissions
\end{itemize}
\end{featurebox}

\begin{examplebox}
\textbf{User Flow Example:}\\
User passes near Banani branch $\rightarrow$ receives notification: ``Stop by Savoy Banani for 10\% off today only!''
\end{examplebox}

\textbf{\color{subsectionTeal}Implementation Tips:} Use background geofencing API with proper battery optimization and user consent.

\textbf{Goal:} Lock in recurring customers with subscription benefits.

\vspace{0.3cm}
\textbf{Core Features:}
\begin{itemize}[leftmargin=*]

    \item Benefits: 3 free scoops + 10\% discount on all orders
    \item Multiple tier levels with increasing benefits

\end{featurebox}

\begin{examplebox}
\textbf{User Flow Example:}\\
User sees promotion: ``Upgrade to Gold Tier — save up to 25\% monthly!''
\end{examplebox}

\textbf{\color{subsectionTeal}Implementation Tips:} Implement recurring payment with SSLCommerz auto-renew API; track subscription status in user profile.

\textbf{Goal:} Expand into high-value use cases like events and catering.

\vspace{0.3cm}

    \item Bulk ordering with custom packages
    \item Catering requests with staff confirmation

\end{featurebox}

\begin{examplebox}
\textbf{User Flow Example:}\\
User books: ``10-person ice-cream party at Dhanmondi Gallery'' for next Saturday.
\end{examplebox}

\textbf{\color{subsectionTeal}Implementation Tips:} Calendar integration + form submission; backend stores event data with staff approval workflow.

\subsection{Multi-language Support (Bangla + English)}
\begin{featurebox}{Feature Overview}
\textbf{Goal:} Achieve national-scale inclusivity with local language support.

\vspace{0.3cm}
\textbf{Core Features:}
\begin{itemize}[leftmargin=*]
    \item Auto-detect phone language settings
    \item Manual language switcher in Settings
    \item Complete UI and content translation
\end{itemize}
\end{featurebox}

\begin{examplebox}
\textbf{User Flow Example:}\\
User switches in Settings $\rightarrow$ entire menu instantly converts to Bangla script.
\end{examplebox}

\textbf{\color{subsectionTeal}Implementation Tips:} Use i18n library (react-i18next or Flutter Intl) with JSON translation files.

\subsection{Advanced Loyalty Tiers}
\begin{featurebox}{Feature Overview}
\textbf{Goal:} Make loyalty program more engaging with tier progression.

\vspace{0.3cm}
\textbf{Core Features:}
\begin{itemize}[leftmargin=*]
    \item Tier levels: Silver, Gold, Platinum
    \item Unlock exclusive offers at each tier

\textbf{\color{subsectionTeal}Implementation Tips:} Tier logic based on yearly spend threshold; trigger tier upgrade celebrations.
